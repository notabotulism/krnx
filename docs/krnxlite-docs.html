Quickstart

Zero config. No Redis. No Docker. Just SQLite under the hood.

from krnxlite import Substrate

# Create a substrate
s = Substrate("my-agent")

# Record events
s.remember("think", {"thought": "User is premium tier"}, from_agent="AGENT")
s.remember("act", {"action": "approve", "amount": 8241}, from_agent="AGENT")
s.remember("result", {"outcome": "loss", "amount": 8141}, from_agent="SYSTEM")

# What went wrong? Query the timeline.
events = s.recall()
bad_decision = events[1]  # The approval

# Branch from BEFORE the bad decision
s.branch("fix", from_event=events[0].event_id)

# Replay with different action
s.remember("act", {"action": "deny"}, from_agent="AGENT", branch="fix")
s.remember("result", {"outcome": "ok", "amount": 0}, from_agent="SYSTEM", branch="fix")

# Two timelines. Same substrate. Cryptographic proof.
assert s.verify()           # main branch intact
assert s.verify("fix")      # fix branch intact
‚úì
That's it. Every event is hash-chained. Every branch is forkable. Every moment is replayable. ~600 lines of Python, zero dependencies.

Core API

Substrate

The core primitive. A content-addressable, append-only event store with branching.

Substrate(workspace: str, path: str = "~/.krnx", enable_hash_chain: bool = True) 
from krnxlite import Substrate

# Default: stores in ~/.krnx/my-agent.db
s = Substrate("my-agent")

# Custom path
s = Substrate("my-agent", path="./data")

# Disable hash chain (faster, no integrity checks)
s = Substrate("my-agent", enable_hash_chain=False)

remember()

Write an event. Returns the event ID.

remember(event_type: str, content: dict, from_agent: str, branch: str = "main") ‚Üí str 
event_id = s.remember(
    "decision",                          # event_type
    {"action": "approve", "confidence": 0.94},  # content
    from_agent="REVIEWER",               # who created it
    branch="main"                         # which timeline
)
# ‚Üí "evt_a1b2c3d4e5f67890"
ParameterTypeDescription
event_typestrCategory: "think", "act", "observe", etc.
contentdictArbitrary JSON-serializable payload
from_agentstrAgent identifier (uppercase by convention)
branchstrTimeline to write to. Default: "main"

recall()

Read events. Returns list of Event objects, most recent last.

recall(limit: int = 100, branch: str = "main") ‚Üí List[Event] 
# Get last 100 events
events = s.recall()

# Get last 10 events from a specific branch
events = s.recall(limit=10, branch="experiment")

# Each event has:
event = events[0]
event.event_id      # "evt_a1b2c3d4..."
event.event_type    # "decision"
event.from_agent    # "REVIEWER"
event.content       # {"action": "approve", ...}
event.timestamp     # 1704067200.123
event.hash          # "8f3a2b1c..."
event.parent_hash   # "7e2b1a0d..." (previous event's hash)

get()

Fetch a single event by ID.

get(event_id: str) ‚Üí Event | None 
# Fetch a specific event
event = s.get("evt_a1b2c3d4e5f67890")

if event:
    print(f"Agent {event.from_agent} did {event.event_type}")
    print(f"Hash: {event.hash}")

search()

Search events by content. Returns matching events.

search(query: str, branch: str = "main", limit: int = 100) ‚Üí List[Event] 
# Find events mentioning "approve"
results = s.search("approve")

# Search in a specific branch
results = s.search("deny", branch="fix")

# Find the bad decision
for event in s.search("amount"):
    print(f"{event.event_id}: {event.content}")

at()

Point-in-time query. Get all events up to a specific timestamp.

at(timestamp: float, branch: str = "main") ‚Üí List[Event] 
import time

# What did we know 1 hour ago?
one_hour_ago = time.time() - 3600
state = s.at(one_hour_ago)

# Reconstruct exact agent state at that moment
for event in state:
    print(f"{event.event_type}: {event.content}")
üí°
This is the "time travel" primitive. Debug exactly what your agent knew when it made a decision.

replay()

Get events in a time range. Useful for replaying a specific window.

replay(start: float, end: float, branch: str = "main") ‚Üí List[Event] 
# Replay the last hour
now = time.time()
events = s.replay(start=now - 3600, end=now)

# Replay a specific incident window
incident_start = 1704067200.0
incident_end = 1704070800.0
events = s.replay(start=incident_start, end=incident_end)

Git Operations

The primitives that make KRNX different.

branch()

Fork the timeline. Create a new branch from any point in history.

branch(name: str, from_event: str = None, from_timestamp: float = None) ‚Üí str 
# Branch from a specific event
s.branch("experiment", from_event="evt_a1b2c3d4")

# Branch from a timestamp
s.branch("rollback", from_timestamp=1704067200.0)

# Branch from current head
s.branch("feature")

# Now write to the new branch
s.remember("test", {"variant": "B"}, from_agent="AGENT", branch="experiment")
‚éá
Counterfactual debugging. Found a bug? Branch from before it happened. Replay with a fix. Compare outcomes.

branches()

List all branches.

branches() ‚Üí List[Dict] 
branches = s.branches()
# [
#   {"name": "main", "parent_branch": None, "fork_event_id": None},
#   {"name": "experiment", "parent_branch": "main", "fork_event_id": "evt_a1b2..."},
#   {"name": "rollback", "parent_branch": "main", "fork_event_id": "evt_c3d4..."},
# ]

verify()

Verify hash-chain integrity. Proves no events were tampered with.

verify(branch: str = "main") ‚Üí bool 
# Verify main branch
assert s.verify()  # True if intact

# Verify a specific branch
assert s.verify("experiment")

# If tampered, raises IntegrityError
from krnxlite import IntegrityError

try:
    s.verify()
except IntegrityError as e:
    print(f"Chain broken at {e.event_id}")
    print(f"Expected: {e.expected_hash}")
    print(f"Got: {e.actual_hash}")
‚õì
Cryptographic proof. Each event's hash includes the previous event's hash. Changing any event breaks the chain.

Agent SDK

Higher-level wrappers for common patterns.

Agent

Track a single agent's behavior with semantic methods.

from krnxlite import Agent

agent = Agent("customer-service")

# Semantic event methods
agent.think("User seems frustrated", confidence=0.8)
agent.hear("I want a refund!", from_="customer")
agent.say("I'd be happy to help with that.")
agent.act("lookup_order", order_id="12345")
agent.observe({"order_status": "delivered"})
agent.result("refund_processed", success=True)

# Query history
timeline = agent.timeline()
thoughts = agent.timeline(type="think")

# Git operations work the same
agent.branch("alt-approach")
assert agent.verify()

Swarm

Multiple agents, one substrate. Full visibility into what everyone did.

from krnxlite import Swarm

swarm = Swarm("code-review")

# Add agents to the shared substrate
security = swarm.agent("security")
performance = swarm.agent("performance")
style = swarm.agent("style")

# Each agent writes to the same timeline
security.act("flag_issue", file="db.py", severity="high")
performance.act("flag_issue", file="api.py", severity="medium")

# Full visibility: see all events or filter by agent
all_events = swarm.timeline()
security_only = swarm.timeline(agent="security")

# Replay what any agent knew at any moment
state = swarm.at(1704067200.0, agent="security")

# Verify the entire substrate
assert swarm.verify()
üëÅ
Visibility, not orchestration. krnx doesn't coordinate agents ‚Äî it gives you a single timeline to see what they all did, when, and why.

mem

Simple key-value style memory. The easiest way to get started.

from krnxlite import mem

# Store anything
mem.store("user said hello")
mem.store({"action": "approve", "confidence": 0.95})

# Get it back
events = mem.all()
recent = mem.last(10)
results = mem.search("approve")

# Time travel
state = mem.at(timestamp)

# Verify
assert mem.verify()
